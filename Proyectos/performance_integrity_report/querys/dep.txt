select c.fec_vcto,  ltrim(rtrim(c.codigo_emi)) as codigo_emi , ltrim(rtrim(c.codigo_fdo)) as codigo_fdo, SUM(c.weight) as weight, c.duration as duration , c.Tasa as tasa,  ltrim(rtrim(c.moneda)) as moneda, SUM(c.monto) as monto 
from ZHIS_Carteras_Recursive as c
INNER JOIN fondosir as f
ON f.Codigo_fdo = c.codigo_fdo collate database_default 
where fecha =[DATE] and Tipo_instrumento ='Deposito' and f.shore= 'onshore' and f.estrategia in ('renta fija' , 'credito', 'retorno absoluto') 
GROUP BY codigo_ins, c.codigo_fdo, duration, tasa, precio, codigo_emi, fec_vcto, c.moneda
