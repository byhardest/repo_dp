select ltrim(rtrim(c.codigo_fdo)) as codigo_fdo, SUM(c.monto) as monto
from ZHIS_Carteras_Recursive as c
INNER JOIN fondosir as f
ON f.Codigo_fdo = c.codigo_fdo collate database_default 
where fecha =[DATE] and f.shore= 'onshore' and f.estrategia in ('renta fija' , 'credito', 'retorno absoluto') 
GROUP BY  c.codigo_fdo
